from tkinter import *
import tkinter.messagebox

root = Tk()
root.title('Northspine Canteen')

frame = Frame(root, width=200, height=200)
frame.pack()

def create_window():
    window = Toplevel(root)

def stall_1_begin():
    stall_1_window = tkinter.messagebox.askquestion('Name of stall 1', 'A delectable selection of Singaporean Classics, served in your very own small wok, would you like to peruse the options?')
    if stall_1_window == 'yes':
        stall_1_interface = Tk()
        stall_1_interface.title('Stall 1')

        def open_stall_1_menu():
            window1 = Toplevel(stall_1_interface)
            food_name1 = Label(window1, text='lmao what food is there even?')
            food_name1.pack()
        stall_1_menu = Button(stall_1_interface,text='Menu for Stall 1', command=open_stall_1_menu)
        stall_1_menu.grid(sticky=W,row=0,column=0, padx=2,pady=2)
        stall_1_menu.grid_rowconfigure(0, weight=1)


        stall_1_waitingtime = Label(stall_1_interface, text='Please enter number of people in the queue here for estimated waiting time.')
        box1 = Entry(stall_1_interface)
        stall_1_waitingtime.grid(row=1,column=0,padx=2,pady=2)
        box1.grid(row=1,column=1)


        stall_1_interface.mainloop()



stall_1 = Button(root, text="Name of stall 1", fg="red", command=stall_1_begin)
stall_1.pack(side=TOP, anchor=N) #why wont it move up!!!!!!!!!!!!!!!!!

root.mainloop()
